Integrating SVGs Into Your Workflow

1. Minify all images, including SVG

GRUNT-CONTRIB-IMAGEMIN

imagemin: {
  app: {
    files: [{
      expand: true,
      cwd: 'src/',
      src: 'assets/svg/*.svg',
      dest: 'build/'
    }]
  }
},

2. Merge a folder of SVGs into a single file with SYMBOLs

GRUNT-CONTRIB-SVGSTORE

svgstore: {
  options: {
    prefix : 'icon-',
    cleanup: ['fill'],
    svg: {
      style : 'width:0;height:0;visibility:hidden;'
    }
  },
  dev: {
    files: {
      '<%= build_dir %>/assets/svg/icons.svg': ['src/assets/svg/*.svg'],
    },
  },
},

3. Add the SVG as the first element in your BODY

Via partials or JavaScript

var request = new XMLHttpRequest();

request.open('GET', 'assets/svg/icons.svg', true);

request.onload = function() {
  if (this.status >= 200 && this.status < 400) {
    $('body').prepend(this.response);
  }
};

request.send();
